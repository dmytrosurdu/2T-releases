\section{Exact Main Clause Ledger}

\begin{enumerate}[leftmargin=2em]
\item \textbf{\(\mathbf{2T}_{\mathrm{rec}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-rec})}.}
  The UOM source is the record/two-arrow/diagonalization package
  Definition~\ref*{uommain-def:uom-limited-scope-package},
  Proposition~\ref*{uommain-prop:uom-limited-scope-realization}, and
  Theorem~\ref*{uommain-thm:uom-equal-source-swap}.
  The broad UOM manuscript presents this at finite cutoff in discrete RG-step form; the appendix
  interface explicitly records that the discrete case is recovered by restriction.

\item \textbf{\(\mathbf{2T}_{\mathrm{det}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-det})}.}
  This detector/nontriviality clause is not exported by the UOM manuscript or the UOM note corpus.
  It belongs to the separate 2T-LS lane.

\item \textbf{\(\mathbf{2T}_{\mathrm{LS}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-LS})}.}
  The appendix definition is
  \[
  \mathbf{2T}_{\mathrm{LS}}
  :=
  \mathbf{2T}_{\mathrm{rec}}+\mathbf{2T}_{\mathrm{det}}.
  \]
  Accordingly, the UOM contribution to \(\mathbf{2T}_{\mathrm{LS}}\) is exactly the
  \(\mathbf{2T}_{\mathrm{rec}}\) source recorded in item \textup{(1)}, while
  \(\mathbf{2T}_{\mathrm{det}}\) remains external to UOM.

\item \textbf{\(\mathbf{2T}_{\mathrm{RG}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-RG})}.}
  The rolled-up manuscript source is item \textbf{(M2)} of
  Definition~\ref*{uommain-def:uom-main-theorem-assumptions}.
  The finite-cutoff UOM realization is
  Theorem~\ref*{uomdesc-thm:uomdesc-2trg}, with the underlying raw descendant package recorded in
  Theorem~\ref*{uomdesc-thm:uomdesc-raw-package}.

\item \textbf{\(\mathbf{2T}_{\mathrm{Peierls}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-cosmo-P})}.}
  The rolled-up manuscript source is item \textbf{(M3)} of
  Definition~\ref*{uommain-def:uom-main-theorem-assumptions}.
  The exact UOM export is Theorem~\ref*{uommain-thm:uom-peierls-package}.

\item \textbf{\(\mathbf{2T}_{\mathrm{EBC}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-EBC})}.}
  The exact UOM export is Theorem~\ref*{uommain-thm:EB-package}.

\item \textbf{\(\mathbf{2T}_{\mathrm{char}}^{\mathrm{weak}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-char-weak})}.}
  The UOM source is Theorem~\ref*{stack-thm:char-weak-interface}.
  Its record-sector provenance inside the broad UOM manuscript is
  Definition~\ref*{uommain-def:pointer-algebra-projection},
  Lemma~\ref*{uommain-lem:tick-fixes-record-algebra},
  Proposition~\ref*{uommain-prop:semiclassical-diag},
  Remark~\ref*{uommain-rem:uom-main-004}, and
  Theorem~\ref*{red-thm:accepted-record-reduction}.
  The stronger concrete export
  Theorem~\ref*{stack-thm:uom-char-export} is a refinement beyond the exact weak clause itself.

\item \textbf{\(\mathbf{2T}_{\mathrm{char}}^{\mathrm{cal}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-char-cal})}.}
  The exact appendix-side invocation is the internal packaging theorem
  Theorem~\ref*{mainapp-thm:char-weak-to-cal} from
  Definition~\ref*{mainapp-def:2T-char-weak}.
  Additional UOM note-level calibration data are
  Proposition~\ref*{stack-prop:minimal-transport-calibration},
  Corollary~\ref*{stack-cor:char-weak-calibration}, and
  Proposition~\ref*{stack-prop:char-calibration-extension}.

\item \textbf{\(\mathbf{2T}_{\mathrm{char}}^{\mathrm{act}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-char-act})}.}
  The UOM source is Theorem~\ref*{stack-thm:char-weak-activity}.
  This uses the Peierls/locality lane exported by Theorem~\ref*{uommain-thm:uom-peierls-package}.

\item \textbf{\(\mathbf{2T}_{\mathrm{alg}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-alg})}.}
  The exact appendix-side invocation is the internal packaging theorem
  Theorem~\ref*{mainapp-thm:char-cal-to-alg}.
  The UOM export is item \textup{(1)} of
  Theorem~\ref*{stack-thm:uom-char-export}, which isolates the characterization algebra
  \[
  \mathcal O^{\mathrm{char}}_{\Lambda_k}
  =
  \mathrm{alg}\{a_{\Lambda_k},S_{\Lambda_k}\}.
  \]

\item \textbf{\(\mathbf{2T}_{\mathrm{char}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-char})}.}
  The appendix defines
  \[
  \mathbf{2T}_{\mathrm{char}}
  :=
  \mathbf{2T}_{\mathrm{char}}^{\mathrm{weak}}
  + \mathbf{2T}_{\mathrm{char}}^{\mathrm{cal}}
  + \mathbf{2T}_{\mathrm{alg}}.
  \]
  Its UOM-side sources are therefore the sources recorded in items \textup{(7)}--\textup{(10)}.
  This aggregate package is defined in the appendix, but it is not itself a separate summand of
  Definition~\ref*{mainapp-def:2T-Main}.

\item \textbf{\(\mathbf{2T}_{\mathrm{sel}}\) and the delayed next-receiver lane.}
  The exact appendix-side same-tick obstruction contract is the chain
  Definition~\ref*{mainapp-def:2T-sel},
  Equation~\ref*{mainapp-eq:selected-line-compatibility},
  Proposition~\ref*{mainapp-prop:Main-selected-line-compatibility}, and
  Theorem~\ref*{mainapp-thm:Main-selected-line-contract-refined}.
  The exact appendix-side delayed next-receiver package is
  Definition~\ref*{mainapp-def:2T-sel-delay},
  Theorem~\ref*{mainapp-thm:Main-delayed-selected-line-contract},
  Theorem~\ref*{mainapp-thm:Main-nonvanishing-from-delayed-lift}, and
  Corollary~\ref*{mainapp-cor:Main-delayed-dominant-kernel-reentry}.
  Upstream UOM supplies the following theorem-backed pieces:
  \begin{enumerate}[label=\textnormal{(\alph*)},leftmargin=2em]
  \item same-tick obstruction through
    Theorem~\ref*{sel-thm:uom-selected-line-no-extra-collapse} and
    Corollary~\ref*{sel-cor:uom-selected-line-quotient-nonzero};
  \item delayed selected-line package through
    Definition~\ref*{delay-def:delayed-selected-line},
    Definition~\ref*{delay-def:delayed-response-map}, and
    Theorem~\ref*{delay-thm:delayed-response-sufficiency}, with explicit imported clause
    \[
    c_{\Lambda_{k+1}}\neq 0,
    \]
    and quotient-first lift into \(\mathfrak D^-_{\Lambda_{k+1}}\) together with the Main-visible
    reduced welded image through
    Theorem~\ref*{splitlift-thm:splitlift-scalar-descendant-lift};
  \item canonical dominant-kernel reentry through the active-response transport note's
    Corollary~\ref*{transport-cor:dominant-kernel-mod-null-reentry};
  \item bridge data through
    Theorem~\ref*{splitlift-thm:splitlift-canonical-quotient-bridge},
    Theorem~\ref*{splitlift-thm:splitlift-canonical-unit-realization}, and
    Corollary~\ref*{splitlift-cor:splitlift-unit-realization-uniqueness}.
    These canonicalize a source-compatible subordinate realization only once one exists.
    What remains open is the derivation of such a realization from branchwise
    \(B\)-localized recoherence alone.
  \end{enumerate}
  Thus the same-tick lane is upstream exact only as an obstruction/no-extra-collapse theorem, while
  the physically correct odd-to-even route is delayed and quotient-first/Main-visible.
  The delayed package reaches the delayed selected line, the descended quotient
  \(\mathfrak D^-_{\Lambda_{k+1}}\), and the Main-visible reduced welded image, but it does
  not by itself identify a canonical microscopic representative on the full source/completion
  class.
  Thus the exact refined same-tick selected-line contract of
  Theorem~\ref*{mainapp-thm:Main-selected-line-contract-refined}
  is downstream packaging, not an upstream UOM theorem on the full microscopic class.

\item \textbf{\(\mathbf{2T}_{\mathrm{conf}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-conf})}.}
  The exact broad-manuscript UOM source is Corollary~\ref*{uommain-cor:uom-core-implies-2T-conf}.
  The ingredients explicitly exported from the semiclassical characterization section are
  Proposition~\ref*{uommain-prop:uom-uv-radial-ward},
  Theorem~\ref*{uommain-thm:uom-uv-dilatation-package}, and
  Theorem~\ref*{uommain-thm:uom-uv-conformal-common-sector}.

\item \textbf{\(\mathbf{2T}_{\mathrm{Main}}\) \textnormal{(Definition~\ref*{mainapp-def:2T-Main})}.}
  The appendix defines
  \[
  \mathbf{2T}_{\mathrm{Main}}
  := 
  \mathbf{2T}_{\mathrm{LS}}
  + \mathbf{2T}_{\mathrm{RG}}
  + \mathbf{2T}_{\mathrm{Peierls}}
  + \mathbf{2T}_{\mathrm{EBC}}
  + \mathbf{2T}_{\mathrm{char}}^{\mathrm{cal}}
  + \mathbf{2T}_{\mathrm{char}}^{\mathrm{act}}
  + \mathbf{2T}_{\mathrm{sel}}
  + \mathbf{2T}_{\mathrm{conf}}.
  \]
  Items \textup{(1)}--\textup{(13)} above record the exact UOM-side source of each imported clause.
  The selected-line clause of item \textup{(12)} is the upstream boundary in that ledger.
\end{enumerate}
